12  package net.jangaroo.smartsprites.maven;
13  
14  import java.io.File;
15  import java.io.IOException;
16  import java.util.ArrayList;
17  import java.util.List;
18  
19  import org.apache.maven.plugin.AbstractMojo;
20  import org.apache.maven.plugin.MojoExecutionException;
21  import org.apache.maven.plugin.MojoFailureException;
22  import org.apache.maven.plugins.annotations.LifecyclePhase;
23  import org.apache.maven.plugins.annotations.Mojo;
24  import org.apache.maven.plugins.annotations.Parameter;
25  import org.carrot2.labs.smartsprites.SmartSpritesParameters;
26  import org.carrot2.labs.smartsprites.SpriteBuilder;
27  import org.carrot2.labs.smartsprites.message.Message;
28  import org.carrot2.labs.smartsprites.message.MessageLog;
29  import org.carrot2.labs.smartsprites.message.PrintStreamMessageSink;

31  /**
32   * Goal which creates Spritesheets from given css and image files Explanation of every variable is taken directly from
33   * the smartsprites documentation (http://csssprites.org/)
34   * <p>
35   * For further information on the use of smartsprites please refer to: http://csssprites.org/
36   */
37  @Mojo(name = "smartsprites", defaultPhase = LifecyclePhase.GENERATE_RESOURCES, requiresProject = false, threadSafe = true)
38  public class SmartSpritesMojo extends AbstractMojo {
39  
40      /** The static root directory mode. */
41      private static final String ROOT_DIR_MODE = "rootDirMode";
42  
43      /** The static css files mode. */
44      private static final String CSS_FILES_MODE = "cssFilesMode";
45  
46      /**
47       * Directory in which SmartSprites processing should be done, required if css-files not specified or if
48       * output-dir-path specified, default: not specified.
49       * <p>
50       * SmartSprites will process all files with the *.css extension found in root-dir-path or any subdirectory of it.
51       * For more fine-grained control over the processed CSS files, see the css-files option.
52       * <p>
53       * If the provided root directory path is relative, it will be resolved against the current working directory.
54       */
55      @Parameter(defaultValue = "${project.basedir}/src/main/sprites", property = "rootDirPath")
56      private File rootDirPath;
57  
58      /**
59       * Paths of CSS files to process, required if root-dir-path not specified, default: not specified.
60       * <p>
61       * SmartSprites will process all CSS files listed using this option. If css-files is to be used together with
62       * output-dir-path, root-dir-path must also be specified so that SmartSprites can preserve the directory structure
63       * found in root-dir-path in output-dir-path. If root-dir-path and output-dir-path are used, css-files outside of
64       * root-dir-path will be ignored.
65       * <p>
66       * Relative CSS file paths provided using this option will be resolved against the current working directory. Please
67       * note that SmartSprites will not expand any wildcards (like style/*.css), it assumes the expansion is performed at
68       * the command line shell level.
69       * <p>
70       * To specify the list of CSS files to process in the SmartSprites Ant task, use one or more nested fileset
71       * elements. Please see the build.xml file in the distribution archive for an example.
72       */
73      @Parameter(property = "cssFiles")
74      private List<File> cssFiles;
75  
76      /**
77       * Output directory for processed CSS files and CSS-relative sprite images, optional, default: not specified.
78       * <p>
79       * If a non-empty output-dir-path is specified, a non-empty root-dir-path must also be provided. The directory
80       * structure relative to the root-dir-path will be preserved in the output directory. E.g. if CSS files are
81       * contained in the css/base directory of root-dir-path, the processed results will be written to
82       * output-dir-path/css/base. Also, CSS-relative sprite images will be written to the output directory. Sprite images
83       * with document-root-relative URLs will be written relative to the document-root-dir-path. If the output-dir-path
84       * directory does not exist, it will be created. If the provided output directory path is relative, it will be
85       * resolved against the current working directory.
86       * <p>
87       * You can leave the output-dir-path empty, in which case the CSS files will be written next to the original CSS
88       * files with the css-file-suffix, and sprite images will be written relative to the original CSS files. If you are
89       * using a non-empty output-dir-path, you might want to use an empty css-file-suffix.
90       */
91      @Parameter(defaultValue = "${project.build.directory}/generated-resources/META-INF/resources/spritesheets", property = "outPutDirPath")
92      private File outputDirPath;
93  
94      /**
95       * Document root path for document-root-relative (starting with /) image urls in CSS, optional, default: not
96       * specified.
97       * <p>
98       * All document-root-relative image and sprite URLs will be taken relative to document-root-dir-path. Also
99       * document-root-relative sprite URLs will be written relative to document-root-dir-path. You can leave this
100      * property empty if your CSS uses only CSS-relative image URLs. If the provided document root directory path is
101      * relative, it will be resolved against the current working directory.
102      */
103     @Parameter(property = "documentRootDirPath")
104     private File documentRootDirPath;
105 
106     /**
107      * Message logging level, optional, default: WARN.
108      * <p>
109      * Messages less important than log-level will not be shown. SmartSprites has 3 levels of log messages (in the
110      * increasing order of importance): INFO: information messages, can be safely ignored IE6NOTICE: notices related to
111      * possible quality loss when creating IE6-friendly sprite images, see also the IE6-friendly PNG option WARN:
112      * warnings related to syntax, IO and sprite rendering quality loss problems that may cause the converted
113      * sprite-based designs look broken
114      */
115     @Parameter(defaultValue = "WARN", property = "logLevel")
116     private String logLevel;
117 
118     /**
119      * Color depth of sprites in the PNG format, optional, default: AUTO. AUTO: PNG color depth will be chosen
120      * automatically. If the sprite image does not contain partial transparencies (alpha channel) and has less than 256
121      * colors, PNG8 will be used. Otherwise, the sprite will be saved in PNG24. DIRECT: PNG sprites will always be saved
122      * in the PNG24 format. INDEXED: PNG sprites will always be saved in the PNG8 format. If the sprite image contains
123      * partial transparencies (alpha channel) or has more than 255 colors, image quality loss may occur and appropriate
124      * warnings will be issued. See also the sprite-matte-color property.
125      */
126     @Parameter(defaultValue = "AUTO", property = "spritePngDepth")
127     private String spritePngDepth;
128 
129     /**
130      * Enables generation of IE6-friendly sprite images, optional, default: disabled.
131      * <p>
132      * If sprite-png-ie6 is specified, for each PNG sprite image with partial transparencies (alpha channel) or more
133      * than 255 colors and any transparencies, SmartSprites will generate a corresponding color-reduced PNG8 file for
134      * IE6.
135      * <p>
136      * An extra IE6-only CSS rule will be added to the generated CSS file to ensure that IE6 (and only IE6) uses the
137      * color-reduced version: #web { width: 17px; height: 17px; background-repeat: no-repeat; background-image:
138      * url('../img/mysprite.png'); -background-image: url('../img/mysprite-ie6.png'); background-position: left -0px; }
139      * <p>
140      * See also the sprite-matte-color property.
141      */
142     @Parameter(defaultValue = "false", property = "spritePngIeSix")
143     private boolean spritePngIeSix;
144 
145     /**
146      * The encoding to assume for input and output CSS files, default: UTF-8. For the list of allowed values, please see
147      * the list of encodings supported in Java.
148      */
149     @Parameter(defaultValue = "UTF-8", property = "cssFileEncoding")
150     private String cssFileEncoding;
151 
152     /**
153      * Suffix to be appended to the processed CSS file name, optional, default: .
154      */
155     @Parameter(defaultValue = "", property = "cssFileSuffix")
156     private String cssFileSuffix;
157 
158     /**
159      * To make sure the different modes are working correctly a mode has to be specified. Modes: - rootDirMode (rootDir
160      * is set, optionally outputdir! No css-Files!) - cssFilesMode (only cssFiles is set, no rootDir, no OutputDir) -
161      * cssFilesWithOutputDirMode (cssFiles are set, as well as rootDir and outputDir)
162      */
163     @Parameter(defaultValue = SmartSpritesMojo.ROOT_DIR_MODE, property = "workingMode")
164     private String workingMode;
165 
166     /**
167      * To enable skipping run of plugin.
168      */
169     @Parameter(defaultValue = "false", alias = "skip", property = "skip")
170     private boolean skip;
171 
172     @Override
173     public void execute() throws MojoExecutionException, MojoFailureException {
174         // Check if plugin run should be skipped
175         if (this.skip) {
176             this.getLog().info("Smartsprites is skipped");
177             return;
178         }
179 
180         // Check for the correct log-level
181         Message.MessageLevel msgLogLevel;
182         try {
183             msgLogLevel = Message.MessageLevel.valueOf(this.logLevel.toUpperCase());
184         } catch (final Exception e) {
185             throw new MojoExecutionException("LogLevel Error - please select a valid value! (INFO, IE6NOTICE, WARN) ",
186                     e);
187         }
188 
189         // Check for the correct PNG-Depth
190         SmartSpritesParameters.PngDepth pngDepth;
191         try {
192             pngDepth = SmartSpritesParameters.PngDepth.valueOf(this.spritePngDepth);
193         } catch (final Exception e) {
194             throw new MojoExecutionException("PngDepth Error - please select a valid value! (AUTO, DIRECT, INDEXED)  ",
195                     e);
196         }
197 
198         // Make sure we are using only correct workingModes
199         if (!(this.workingMode.equals(SmartSpritesMojo.ROOT_DIR_MODE)
200                 || this.workingMode.equals(SmartSpritesMojo.CSS_FILES_MODE)
201                 || this.workingMode.equals("cssFilesWithOutputDirMode"))) {
202             throw new MojoExecutionException(
203                     "workingMode Error - plese select a valid value! (rootDirMode, cssFilesMode, cssFilesWithOutputDirMode)");
204         }
205 
206         // Variables for paths
207         String rootDirPathTemp = "";
208         String outputDirPathTemp = "";
209         String documentRootDirPathTemp = "";
210         List<String> cssFilesTemp = new ArrayList<>();
211 
212         // Check if we should set cssFiles to null, or if we are in a working mode
213         // where we want to use cssFiles
214         if (!this.workingMode.equals(SmartSpritesMojo.ROOT_DIR_MODE)) {
215             for (final File cssFile : this.cssFiles) {
216                 if (cssFile.exists()) {
217                     cssFilesTemp.add(cssFile.toString());
218                 } else {
219                     throw new MojoExecutionException("The following css-file doesn't exist: " + cssFile);
220                 }
221             }
222         } else {
223             this.cssFiles = null;
224         }
225 
226         // Check if the folders exist or are at least configured
227         if (this.rootDirPath != null && !this.workingMode.equals(SmartSpritesMojo.CSS_FILES_MODE)) {
228             if (this.rootDirPath.exists()) {
229                 rootDirPathTemp = this.rootDirPath.toString();
230             } else {
231                 throw new MojoExecutionException("The rootDirPath doesn't exist. " + this.rootDirPath.toString());
232             }
233         }
234 
235         // Check if the outputDirPath is set
236         if (this.outputDirPath != null && !this.workingMode.equals(SmartSpritesMojo.CSS_FILES_MODE)) {
237             outputDirPathTemp = this.outputDirPath.toString();
238         }
239 
240         // Check if the documentRootDirPath is set
241         if (this.documentRootDirPath != null) {
242             documentRootDirPathTemp = this.documentRootDirPath.toString();
243         }
244 
245         // Determine which workingMode we are in and check if all conditions for that
246         // specific mode are matched
247         if (this.workingMode.equals(SmartSpritesMojo.ROOT_DIR_MODE)) {
248             if (this.rootDirPath == null) {
249                 throw new MojoExecutionException("Please configure a rootDirPath.");
250             }
251             cssFilesTemp = null;
252         } else if (this.workingMode.equals(SmartSpritesMojo.CSS_FILES_MODE)) {
253             if (this.cssFiles == null) {
254                 throw new MojoExecutionException("Please configure some cssFiles.");
255             }
256             rootDirPathTemp = null;
257             outputDirPathTemp = null;
258         } else if (this.workingMode.equals("cssFilesWithOutputDirMode") && this.cssFiles == null
259                 || this.rootDirPath == null || this.outputDirPath == null) {
260             throw new MojoExecutionException("Please configure cssFiles and/or a rootDirPath and/or an outputDirPath");
261         }
262 
263         // Configure the SmartSpritesParameters for execution
264         SmartSpritesParameters smartParameters = new SmartSpritesParameters(rootDirPathTemp, cssFilesTemp,
265                 outputDirPathTemp, documentRootDirPathTemp, msgLogLevel, this.cssFileSuffix, pngDepth,
266                 this.spritePngIeSix, this.cssFileEncoding);
267 
268         final MessageLog messageLog = new MessageLog(
269                 new PrintStreamMessageSink(System.out, smartParameters.getLogLevel()));
270         SpriteBuilder spriteBuilder;
271 
272         // Try to execute SmartSprites with the configured parameters and the defined messageLog
273         try {
274             spriteBuilder = new SpriteBuilder(smartParameters, messageLog);
275             spriteBuilder.buildSprites();
276         } catch (final IOException e) {
277             throw new MojoExecutionException("Smartsprites error: ", e);
278         }
279     }
